23 /// Checks whether two floating point numbers x, y are close to each other.
24 ///
25 /// It outputs: min(|x-y|, |x-y| / max(|x|, |y|))
26 ///
27 /// The output is between 0.0 and 2.0:
28 ///
29 /// * x and y are close => ~0.0
30 /// * x and y are NOT close and
31 /// - |x| and |y| are NOT close => ~1.0
32 /// - |x| and |y| are close => ~2.0
33 inline double relativeCloseness(double x, double y) {
34  // https://c-faq.com/fp/fpequal.html
35  double abs_diff = std::abs(x - y);
36  double rel_diff = abs_diff / std::max(std::abs(x), std::abs(y));
37 
38  // We are using "rel_diff" from the c-faq reference. However, instead of just
39  // checking for x==0 and y==0, we perform the minimum over `rel_diff` and
40  // `abs_diff`. This is to avoid surprises when there are two small numbers
41  // with opposite sign.
42  // E.g. for x=0.001 and y=-0.001 we have abs_diff = 0.002, but rel_diff
43  // = 2.0, hence relativeCloseness(0.001, -0.001) = 0.002.
44  return std::min(abs_diff, rel_diff);
45 }

51 template <class TPoint, class TPoint2>
52 struct CheckNear<
53  TPoint,
54  TPoint2,

56  std::is_arithmetic_v<TPoint> && std::is_arithmetic_v<TPoint2>,
57  void>> {
58  static Expected<Success> check(
59  bool check_relative,
60  double const& lhs,
61  char const* lhs_cstr,
62  double const& rhs,
63  char const* rhs_cstr,
64  double const& thr) {
65  double err =
66  check_relative ? relativeCloseness(rhs, lhs) : std::abs(rhs - lhs);
67  if (!(err <= thr)) {
68  // inverted comparison, so we also end up here if err or max_error

71  "LHS: `{}` is:\n"
72  "{}\n\n"
73  "RHS `{}` is:\n"
74  "{}\n\n",
75  lhs_cstr,
76  rhs_cstr,
77  err,
78  thr,
79  lhs_cstr,
80  lhs,
81  rhs_cstr,
82  rhs);
83  }
84  return Success{};
85  }

88 template <class TScalar, int kRows, int kCols>
89 struct CheckNear<
90  Eigen::Matrix<TScalar, kRows, kCols>,
91  Eigen::Matrix<TScalar, kRows, kCols>,
92  std::enable_if_t<std::is_arithmetic_v<TScalar>, void>> {
93  static Expected<Success> check(
94  bool check_relative,
95  Eigen::Matrix<TScalar, kRows, kCols> const& lhs,
96  char const* lhs_cstr,
97  Eigen::Matrix<TScalar, kRows, kCols> const& rhs,
98  char const* rhs_cstr,
99  double const& thr) {
100  double max_error = 0.0;
101  int max_row_idx = -1;
102  int max_col_idx = -1;
103 
104  auto nearness_fn =
105  check_relative
106  ? [](double x, double y) { return relativeCloseness(x, y); }
107  : [](double x, double y) { return std::abs(x - y); };
108 
109  for (int c = 0; c < lhs.cols(); ++c) {
110  for (int r = 0; r < lhs.rows(); ++r) {
111  double err = nearness_fn(lhs(r, c), rhs(r, c));
112 
113  if (!(err <= max_error)) {
114  // inverted comparison, so we also end up here if err or max_error
115  // are NAN
116  max_error = err;
117  max_row_idx = r;
118  max_col_idx = c;
119  }
120  }
121  }
122  if (max_error < thr) {
123  // all errors below threshold
124  return Success{};
125  }
126  return FARM_UNEXPECTED(
127  "Not true: {}[{}, {}] near {}[{}, {}]; has error of {} (thr: {})\n"
128  "LHS `{}` is:\n"
129  "{}\n\n"
130  "RHS `{}` is:\n"
131  "{}\n\n",
132  lhs_cstr,
133  max_row_idx,
134  max_col_idx,
135  rhs_cstr,
136  max_row_idx,
137  max_col_idx,
138  max_error,
139  thr,
140  lhs_cstr,
141  lhs,
142  rhs_cstr,
143  rhs);
144  }
145 };
