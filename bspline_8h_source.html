27  [[nodiscard]] Vector interpolate(int i, double u) const {
28  SOPHUS_ASSERT_GE(i, 0);
29  SOPHUS_ASSERT_LE(i, this->getNumSegments());
30 
31  details::SegmentCase segment_case =
32  i == 0
33  ? details::SegmentCase::first
34  : (i == this->getNumSegments() - 1 ? details::SegmentCase::last
35  : details::SegmentCase::normal);
36 
37  int idx_prev = std::max(0, i - 1);
38  int idx_0 = i;
39  int idx_1 = i + 1;
40  int idx_2 = std::min(i + 2, int(this->control_points_.size()) - 1);
41 
42  return details::CubicBSplineSegment<Scalar, kDim>(
43  segment_case,
44  control_points_[idx_prev],
45  control_points_[idx_0],
46  control_points_[idx_1],
47  control_points_[idx_2])
48  .interpolate(u);
49  }

51  [[nodiscard]] Eigen::Matrix<Scalar, kDim, kDim> dxiInterpolate(
52  int i, double u, int control_point_idx) const {
53  SOPHUS_ASSERT_GE(i, 0);
54  SOPHUS_ASSERT_LE(i, this->getNumSegments());
55 
56  details::SegmentCase segment_case =
57  i == 0
58  ? details::SegmentCase::first
59  : (i == this->getNumSegments() - 1 ? details::SegmentCase::last
60  : details::SegmentCase::normal);
61 
62  int idx_prev = std::max(0, i - 1);
63  int idx_0 = i;
64  int idx_1 = i + 1;
65  int idx_2 = std::min(i + 2, int(this->control_points_.size()) - 1);
66 
67  details::CubicBSplineSegment<Scalar, kDim> spline_segment(
68  segment_case,
69  control_points_[idx_prev],
70  control_points_[idx_0],
71  control_points_[idx_1],
72  control_points_[idx_2]);
73 
74  Eigen::Matrix<Scalar, kDim, kDim> dxi;
75  dxi.setZero();
76 
77  if (idx_prev == control_point_idx) {
78  dxi += spline_segment.dxiInterpolate(u, 0);
79  }
80  if (idx_0 == control_point_idx) {
81  dxi += spline_segment.dxiInterpolate(u, 1);
82  }
83  if (idx_1 == control_point_idx) {
84  dxi += spline_segment.dxiInterpolate(u, 2);
85  }
86  if (idx_2 == control_point_idx) {
87  dxi += spline_segment.dxiInterpolate(u, 3);
88  }
89  return dxi;
90  }

110 class CubicBSpline {
111  public:
112  using Scalar = TScalar;
113  using Vector = Eigen::Vector<Scalar, kDim>;
114 
115  CubicBSpline(std::vector<Vector> control_points, double t0, double delta_t)
116  : impl_(std::move(control_points), delta_t), t0_(t0) {}
117 
118  [[nodiscard]] Vector interpolate(double t) const {
119  SegmentCoordinate index_and_u = this->indexAndU(t);
120  return impl_.interpolate(index_and_u.segment_idx, index_and_u.fraction);
121  }
122 
123  [[nodiscard]] Eigen::Matrix<Scalar, kDim, kDim> dxiInterpolate(
124  double t, int control_point_idx) const {
125  SegmentCoordinate index_and_u = this->indexAndU(t);
126  return impl_.dxiInterpolate(
127  index_and_u.segment_idx, index_and_u.fraction, control_point_idx);
128  }
129 
130  [[nodiscard]] double t0() const { return t0_; }
131 
132  [[nodiscard]] double tmax() const {
133  return t0_ + impl_.deltaT() * getNumSegments();
134  }
135 
136  [[nodiscard]] std::vector<Vector> const& controlPoints() const {
137  return impl_.controlPoints();
138  }
139 
140  std::vector<Vector>& controlPoints() { return impl_.controlPoints(); }
141 
142  [[nodiscard]] int getNumSegments() const { return impl_.getNumSegments(); }
143 
144  [[nodiscard]] double s(double t) const { return (t - t0_) / impl_.deltaT(); }
145 
146  [[nodiscard]] double deltaT() const { return impl_.deltaT(); }
147 
148  [[nodiscard]] SegmentCoordinate indexAndU(double t) const {
149  SOPHUS_ASSERT_GE(t, t0_);
150  SOPHUS_ASSERT_LE(t, this->tmax());
151 
152  double s = this->s(t);
153  double segment_idx = NAN;
154  SegmentCoordinate index_and_u;
155  index_and_u.fraction = std::modf(s, &segment_idx);
156  index_and_u.segment_idx = int(segment_idx);
157  if (index_and_u.fraction > sophus::kEpsilonF64) {
158  return index_and_u;
159  }
160 
161  // u ~=~ 0.0
162  if (index_and_u.segment_idx < getNumSegments() / 2) {
163  // First half of spline, keep as is (i, 0.0).
164  return index_and_u;
165  }
166  // Second half of spline, use (i-1, 1.0) instead. This way we can
167  // represent t == tmax (and not just t<tmax).
168  index_and_u.fraction += 1.0;
169  --index_and_u.segment_idx;
170 
171  return index_and_u;
172  }
173 
174  private:
175  CubicBSplineImpl<Scalar, kDim> impl_;
176 
177  double t0_;
178 };
