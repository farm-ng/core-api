16 /// Generic complex number implementation
17 ///
18 /// Impl class without any storage, but only static methods.
19 template <class TScalar>
20 class ComplexImpl {
21  public:
22  /// The underlying scalar type.
23  using Scalar = TScalar;
24  /// A complex number is a tuple.
25  static int constexpr kNumParams = 2;
26  /// Complex multiplication is commutative (and it is a field itself).
27  static bool constexpr kIsCommutative = true;
28 
29  /// Parameter storage type - to be passed to static functions.
30  using Params = Eigen::Vector<Scalar, kNumParams>;
31 
32  /// Return type of binary operator
33  ///
34  /// In particular relevant mixing different scalars (such as double and
35  /// ceres::Jet).
36  template <class TCompatibleScalar>
37  using ParamsReturn = Eigen::Vector<
38  typename Eigen::ScalarBinaryOpTraits<Scalar, TCompatibleScalar>::
39  ReturnType,
40  2>;
41 
42  // factories
43 
44  /// Returns zero constant which is (0, 0).
45  static auto zero() -> Eigen::Vector<Scalar, 2> {
46  return Eigen::Vector<Scalar, 2>::Zero();
47  }
48 
49  /// Returns one constant which is (1, 0).
50  static auto one() -> Eigen::Vector<Scalar, 2> {
51  return Eigen::Vector<Scalar, 2>(1.0, 0.0);
52  }

54  /// There are no particular constraints on the complex space and this function

56  static auto areParamsValid(Params const& /*unused*/)

73  /// Complex addition given two complex numbers (a,i) and (b,j).
74  ///
75  /// Returns (a+b, i+j).
76  template <class TCompatibleScalar>
77  static auto addition(
78  Eigen::Vector<Scalar, 2> const& lhs_real_imag,
79  Eigen::Vector<TCompatibleScalar, 2> const& rhs_real_imag)
80  -> ParamsReturn<TCompatibleScalar> {
81  return lhs_real_imag + rhs_real_imag;
82  }
83 
84  /// Complex multiplication given two complex numbers (a,i) and (b,j).
85  ///
86  /// Returns (a+b-i*j, a*j+b*i).
87  template <class TCompatibleScalar>
88  static auto multiplication(
89  Eigen::Vector<Scalar, 2> const& lhs_real_imag,
90  Eigen::Vector<TCompatibleScalar, 2> const& rhs_real_imag)
91  -> ParamsReturn<TCompatibleScalar> {
92  // complex multiplication
93  return ParamsReturn<TCompatibleScalar>(
94  lhs_real_imag.x() * rhs_real_imag.x() -
95  lhs_real_imag.y() * rhs_real_imag.y(),
96  lhs_real_imag.x() * rhs_real_imag.y() +
97  lhs_real_imag.y() * rhs_real_imag.x());
98  }
99 
100  /// Given input (a,i), return complex conjugate (a, -i).
101  static auto conjugate(Eigen::Vector<Scalar, 2> const& a)
102  -> Eigen::Vector<Scalar, 2> {
103  return Eigen::Vector<Scalar, 2>(a.x(), -a.y());
104  }
105 
106  /// Given input (a,i) return complex inverse (a, -i) / (a**2 + i**2).
107  static auto inverse(Eigen::Vector<Scalar, 2> const& real_imag)
108  -> Eigen::Vector<Scalar, 2> {
109  return conjugate(real_imag) / squaredNorm(real_imag);
110  }
111 
112  /// Given input (a,i), returns complex norm sqrt(a**2 + i**2).
113  static auto norm(Eigen::Vector<Scalar, 2> const& real_imag) -> Scalar {
114  using std::hypot;
115  return hypot(real_imag.x(), real_imag.y());
116  }
117 
118  /// Given input (a,i), returns square of complex norm a**2 + i**2.
119  static auto squaredNorm(Eigen::Vector<Scalar, 2> const& real_imag) -> Scalar {
120  return real_imag.squaredNorm();
121  }
122 };

124 /// Complex number class template.
125 ///
126 /// A complex number is manifold with additional structure. In particular,
127 /// it is - like the quaternion numbers - a division ring and fulfills the
128 /// sophus::concepts::DivisionRingConcept. Furthermore, it is has commutative
129 /// multiplication (as opposed to the quaternion numbers) and hence is a field.
130 /// The complex numbers are the only other field of real numbers in addition to
131 /// the real numbers itself.
132 template <class TScalar>
133 class Complex {
134  public:
135  /// The underlying scalar type.
136  using Scalar = TScalar;
137  /// Type of the imaginary part is also a Scalar and hence has the some type
138  /// than the real part.
139  using Imag = Scalar;
140  /// The stateless implementation.
141  using Impl = ComplexImpl<Scalar>;
142 
143  /// A complex number is a tuple.
144  static int constexpr kNumParams = 2;
145 
146  /// Parameter storage type.
147  using Params = Eigen::Vector<Scalar, kNumParams>;
148 
149  /// Return type of binary operator
150  ///
151  /// In particular relevant mixing different scalars (such as double and
152  /// ceres::Jet).
153  template <class TCompatibleScalar>
154  using ComplexReturn =
155  Complex<typename Eigen::ScalarBinaryOpTraits<Scalar, TCompatibleScalar>::
156  ReturnType>;
157 
158  // constructors and factories
159 
160  /// Default constructor creates a zero complex number (0, 0).
161  Complex() : params_(Impl::zero()) {}
162 
163  /// Copy constructor
164  Complex(Complex const&) = default;
165  /// Copy assignment
166  auto operator=(Complex const&) -> Complex& = default;
167 
168  /// Returns zero complex number (0, 0).
169  static auto zero() -> Complex { return Complex::fromParams(Impl::zero()); }
170 
171  /// Returns identity complex number (1, 0).
172  static auto one() -> Complex { return Complex::fromParams(Impl::one()); }
173 
174  /// Constructs complex number from params tuple.
175  static auto fromParams(Params const& params) -> Complex {
176  Complex z(UninitTag{});
177  z.setParams(params);
178  return z;
179  }
180 
181  /// Returns params tuple.
182  [[nodiscard]] auto params() const -> Params const& { return params_; }
183 
184  /// Set state using params tuple.
185  void setParams(Params const& params) { params_ = params; }
186 
187  /// Returns reference to real scalar.
188  auto real() -> Scalar& { return params_[0]; }
189  /// Returns reference to real scalar.
190  [[nodiscard]] auto real() const -> Scalar const& { return params_[0]; }
191 
192  /// Returns reference to imaginary scalar.
193  auto imag() -> Scalar& { return params_[1]; }
194  /// Returns reference to imaginary scalar.
195  [[nodiscard]] auto imag() const -> Scalar const& { return params_[1]; }
196 
197  /// Complex addition
198  template <class TCompatibleScalar>
199  auto operator+(Complex<TCompatibleScalar> const& other) const
200  -> ComplexReturn<TCompatibleScalar> {
201  return Complex::fromParams(Impl::addition(this->params_, other.params()));
202  }
203 
204  /// Complex multiplication
205  template <class TCompatibleScalar>
206  auto operator*(Complex<TCompatibleScalar> const& other) const
207  -> ComplexReturn<TCompatibleScalar> {
208  return ComplexReturn<TCompatibleScalar>::fromParams(
209  Impl::multiplication(this->params_, other.params()));
210  }
211 
212  /// Complex conjugate
213  [[nodiscard]] auto conjugate() const -> Complex {
214  return Complex::fromParams(Impl::conjugate(this->params_));
215  }
216 
217  /// Complex inverse
218  [[nodiscard]] auto inverse() const -> Complex {
219  return Complex::fromParams(Impl::inverse(this->params_));
220  }
221 
222  /// Complex norm
223  [[nodiscard]] auto norm() const -> Scalar {
224  return Impl::norm(this->params_);
225  }
226 
227  /// Square of complex norm
228  [[nodiscard]] auto squaredNorm() const -> Scalar {
229  return Impl::squaredNorm(this->params_);
230  }
231 
232  private:
233  Complex(UninitTag /*unused*/) {}
234  Eigen::Vector2<Scalar> params_;
235 };
