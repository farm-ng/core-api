21 template <concepts::LieGroup TGroup>
22 class CubicLieGroupBSplineFn {
23  public:
24  using LieGroup = TGroup;
25  static int const kDof = LieGroup::kDof;
26  using Scalar = typename LieGroup::Scalar;
27  using Transformation =
28  Eigen::Matrix<Scalar, TGroup::kAmbientDim, TGroup::kAmbientDim>;
29  using Tangent = typename LieGroup::Tangent;
30 
31  static LieGroup parentFromSpline(
32  LieGroup const& parent_ts_control_point,
33  std::tuple<Tangent, Tangent, Tangent> const& control_tangent_vectors,
34  double u) {
35  auto aa = a(control_tangent_vectors, u);
36  return parent_ts_control_point * std::get<0>(aa) * std::get<1>(aa) *
37  std::get<2>(aa);
38  }
39 
40  static Transformation dtParentFromSpline(
41  LieGroup const& parent_ts_control_point,
42  std::tuple<Tangent, Tangent, Tangent> const& control_tangent_vectors,
43  double u,
44  double delta_t) {
45  auto aa = a(control_tangent_vectors, u);
46  auto dt_aa = dtA(aa, control_tangent_vectors, u, delta_t);
47  return parent_ts_control_point.matrix() *
48  ((std::get<0>(dt_aa) * std::get<1>(aa).matrix() *
49  std::get<2>(aa).matrix()) +
50  (std::get<0>(aa).matrix() * std::get<1>(dt_aa) *
51  std::get<2>(aa).matrix()) +
52  (std::get<0>(aa).matrix() * std::get<1>(aa).matrix() *
53  std::get<2>(dt_aa)));
54  }
55 
56  static Transformation dt2ParentFromSpline(
57  LieGroup const& parent_ts_control_point,
58  std::tuple<Tangent, Tangent, Tangent> const& control_tangent_vectors,
59  double u,
60  double delta_t) {
61  using TScalar = typename LieGroup::Scalar;
62  auto aa = a(control_tangent_vectors, u);
63  auto dt_aa = dtA(aa, control_tangent_vectors, u, delta_t);
64  auto dt2_aa = dt2A(aa, dt_aa, control_tangent_vectors, u, delta_t);
65 
66  return parent_ts_control_point.matrix() *
67  ((std::get<0>(dt2_aa) * std::get<1>(aa).matrix() *
68  std::get<2>(aa).matrix()) +
69  (std::get<0>(aa).matrix() * std::get<1>(dt2_aa) *
70  std::get<2>(aa).matrix()) +
71  (std::get<0>(aa).matrix() * std::get<1>(aa).matrix() *
72  std::get<2>(dt2_aa)) +
73  TScalar(2) * ((std::get<0>(dt_aa) * std::get<1>(dt_aa) *
74  std::get<2>(aa).matrix()) +
75  (std::get<0>(dt_aa) * std::get<1>(aa).matrix() *
76  std::get<2>(dt_aa)) +
77  (std::get<0>(aa).matrix() * std::get<1>(dt_aa) *
78  std::get<2>(dt_aa))));
79  }
80 
81  static std::tuple<LieGroup, LieGroup, LieGroup> a(
82  std::tuple<Tangent, Tangent, Tangent> const& control_tangent_vectors,
83  double u) {
84  Eigen::Vector3d b = details::CubicBSplineBasis<double>::b(u);
85  return std::make_tuple(
86  LieGroup::exp(b[0] * std::get<0>(control_tangent_vectors)),
87  LieGroup::exp(b[1] * std::get<1>(control_tangent_vectors)),
88  LieGroup::exp(b[2] * std::get<2>(control_tangent_vectors)));
89  }
90 
91  static Transformation dx1A(
92  Tangent const& control_tangent_vectors, double u, int i) {
93  Eigen::Vector3d b = details::CubicBSplineBasis<double>::b(u);
94  Eigen::Vector<Scalar, kDof> t;
95  t[i] = Scalar(1.0);
96  return LieGroup::hat(t) * LieGroup::exp(b[0] * t).matrix();
97  }
98 
99  static std::tuple<Transformation, Transformation, Transformation> dtA(
100  std::tuple<LieGroup, LieGroup, LieGroup> const& aa,
101  std::tuple<Tangent, Tangent, Tangent> const& control_tangent_vectors,
102  double u,
103  double delta_t) {
104  Eigen::Vector3d dt_b = details::CubicBSplineBasis<double>::dtB(u, delta_t);
105  return std::make_tuple(
106  dt_b[0] * std::get<0>(aa).matrix() *
107  LieGroup::hat(std::get<0>(control_tangent_vectors)),
108  dt_b[1] * std::get<1>(aa).matrix() *
109  LieGroup::hat(std::get<1>(control_tangent_vectors)),
110  dt_b[2] * std::get<2>(aa).matrix() *
111  LieGroup::hat(std::get<2>(control_tangent_vectors)));
112  }
113 
114  static std::tuple<Transformation, Transformation, Transformation> dt2A(
115  std::tuple<LieGroup, LieGroup, LieGroup> const& aa,
116  std::tuple<Transformation, Transformation, Transformation> const& dt_aa,
117  std::tuple<Tangent, Tangent, Tangent> const& control_tangent_vectors,
118  double u,
119  double delta_t) {
120  Eigen::Vector3d dt_b = details::CubicBSplineBasis<double>::dtB(u, delta_t);
121  Eigen::Vector3d dt2_b =
122  details::CubicBSplineBasis<double>::dt2B(u, delta_t);
123 
124  return std::make_tuple(
125  (dt_b[0] * std::get<0>(dt_aa).matrix() +
126  dt2_b[0] * std::get<0>(aa).matrix()) *
127  LieGroup::hat(std::get<0>(control_tangent_vectors)),
128  (dt_b[1] * std::get<1>(dt_aa).matrix() +
129  dt2_b[1] * std::get<1>(aa).matrix()) *
130  LieGroup::hat(std::get<1>(control_tangent_vectors)),
131  (dt_b[2] * std::get<2>(dt_aa).matrix() +
132  dt2_b[2] * std::get<2>(aa).matrix()) *
133  LieGroup::hat(std::get<2>(control_tangent_vectors)));
134  }
135 };

268 template <concepts::LieGroup TGroup>
269 class CubicLieGroupBSpline {
270  public:
271  using LieGroup = TGroup;
272  using Scalar = typename LieGroup::Scalar;
273  using Transformation =
274  Eigen::Matrix<Scalar, TGroup::kAmbientDim, TGroup::kAmbientDim>;
275  using Tangent = typename LieGroup::Tangent;
276 
277  CubicLieGroupBSpline(
278  std::vector<LieGroup> parent_ts_control_point, double t0, double delta_t)
279  : impl_(std::move(parent_ts_control_point), delta_t), t0_(t0) {}
280 
281  [[nodiscard]] LieGroup parentFromSpline(double t) const {
282  SegmentCoordinate index_and_u = this->indexAndU(t);
283 
284  return impl_.parentFromSpline(
285  index_and_u.segment_idx, index_and_u.fraction);
286  }
287 
288  [[nodiscard]] Transformation dtParentFromSpline(double t) const {
289  SegmentCoordinate index_and_u = this->indexAndU(t);
290  return impl_.dtParentFromSpline(
291  index_and_u.segment_idx, index_and_u.fraction);
292  }
293 
294  [[nodiscard]] Transformation dt2ParentFromSpline(double t) const {
295  SegmentCoordinate index_and_u = this->indexAndU(t);
296  return impl_.dt2ParentFromSpline(
297  index_and_u.segment_idx, index_and_u.fraction);
298  }
299 
300  [[nodiscard]] double t0() const { return t0_; }
301 
302  [[nodiscard]] double tmax() const {
303  return t0_ + impl_.deltaT() * getNumSegments();
304  }
305 
306  [[nodiscard]] std::vector<LieGroup> const& parentFromsControlPoint() const {
307  return impl_.parentFromsControlPoint();
308  }
309 
310  std::vector<LieGroup>& parentFromsControlPoint() {
311  return impl_.parentFromsControlPoint();
312  }
313 
314  [[nodiscard]] int getNumSegments() const { return impl_.getNumSegments(); }
315 
316  [[nodiscard]] double s(double t) const { return (t - t0_) / impl_.deltaT(); }
317 
318  [[nodiscard]] double deltaT() const { return impl_.deltaT(); }
319 
320  [[nodiscard]] SegmentCoordinate indexAndU(double t) const {
321  SOPHUS_ASSERT(t >= t0_, "{} vs. {}", t, t0_);
322  SOPHUS_ASSERT(t <= this->tmax(), "{} vs. {}", t, this->tmax());
323 
324  double s = this->s(t);
325  double i = NAN;
326  SegmentCoordinate index_and_u;
327  index_and_u.fraction = std::modf(s, &i);
328  index_and_u.segment_idx = int(i);
329  if (index_and_u.fraction > sophus::kEpsilonF64) {
330  return index_and_u;
331  }
332 
333  // u ~=~ 0.0
334  if (index_and_u.segment_idx < getNumSegments() / 2) {
335  // First half of spline, keep as is (i, 0.0).
336  return index_and_u;
337  }
338  // Second half of spline, use (i-1, 1.0) instead. This way we can
339  // represent t == tmax (and not just t<tmax).
340  index_and_u.fraction += 1.0;
341  --index_and_u.segment_idx;
342 
343  return index_and_u;
344  }
345 
346  private:
347  CubicLieGroupBSplineImpl<LieGroup> impl_;
348 
349  double t0_;
350 };
