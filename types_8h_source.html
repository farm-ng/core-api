33 /// Condition for clear some or all curve data
34 ///
35 /// Note: The predicate is applied before new data is added hence if e.g.
36 /// we have a curve struct with `curve.max_x_range == 0` pre-exiting data is
37 /// erased from the plot and then filled with `curve.x_y_pairs`.
38 struct ClearCondition {
39  /// Maximal range of x values of graph. Here, x_range = max(X) - min(X).
40  double max_x_range = 10000.0;
41 };
42 
43 /// Creates clear condition for new curve. All preexisting data will be cleared.
44 inline ClearCondition addNew() { return ClearCondition{.max_x_range = 0}; }

78 // Add a "3-vec curve" to the plot or append to exiting one with path being
79 // "plot_name/curve_name". If a curve with that name already exists, it will be
80 // overwritten.
81 //
82 // A "3-vec curve" are 3 curves described a list of (3+1)-dimensional tuples
83 // `x_vec_pairs`, where the first component of the tuples is the x coordinate
84 // for each curve, and the remaining components are the y coordinates.
