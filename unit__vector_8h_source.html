33 /// A n-dimensional vector of unit length.
34 ///
35 /// Impl class without any storage, but only static methods.
36 template <class TScalar, int kDim>
37 class UnitVectorImpl {
38  public:
39  /// scalar type
40  using Scalar = TScalar;
41 
42  /// The DOF is one dimension smaller than the embedding dimension.
43  /// Examples:
44  /// - A unit circle lives in a 2d space, but a point on a circle can be
45  /// represented using a single scalar (e.g. an angle).
46  /// - A unit sphere lives in a 3d space, but a point on a sphere can be
47  /// described using a 2-tuple (e.g. latitude and longitude).
48  static int constexpr kDof = kDim - 1;
49  /// Internally the unit vector is represented by a vector, hence the number of
50  /// parameters equals the number of dimension of the embedding space.
51  static int constexpr kNumParams = kDim;
52 
53  /// internal parameter representation
54  using Params = Eigen::Vector<Scalar, kNumParams>;
55  /// tangent vector representation
56  using Tangent = Eigen::Vector<Scalar, kDof>;
57 
58  /// return the unit vector along x: (1, 0, 0, ...)
59  static auto unitX() -> UnitVector<Scalar, kDim> {
60  return UnitVector<Scalar, kDim>::fromUnitVector(Params::UnitX());
61  }
62 
63  /// return the unit vector along y: (0, 1, 0, 0, ...)
64  ///
65  /// Precondition: kDim >= 1
66  static auto unitY() -> UnitVector<Scalar, kDim> {
67  return UnitVector<Scalar, kDim>::fromUnitVector(Params::UnitY());
68  }
69 
70  /// return the unit vector along z: (0, 0, 1, 0, 0, ...)
71  ///
72  /// Precondition: kDim >= 2
73  static auto unitZ() -> UnitVector<Scalar, kDim> {
74  return UnitVector<Scalar, kDim>::fromUnitVector(Params::UnitZ());
75  }
76 
77  /// Returns false if the norm of the vector if not close to 1.
78  static auto areParamsValid(Params const& unit_vector)
79  -> sophus::Expected<Success> {
80  static Scalar const kThr = kEpsilon<Scalar>;
81  Scalar const squared_norm = unit_vector.squaredNorm();
82  using std::abs;
83  if (!(abs(squared_norm - 1.0) <= kThr)) {
84  return SOPHUS_UNEXPECTED(
85  "unit vector ({}) is not of unit length.\n"
86  "Squared norm: {}, thr: {}",
87  unit_vector.transpose(),
88  squared_norm,
89  kThr);
90  }
91  return sophus::Expected<Success>{};
92  }
93 
94  /// oplus operator
95  ///
96  /// See: Hartley Zisserman
97  static auto oplus(Params const& params, Tangent const& delta) -> Params {
98  return matRx(params) * exp(delta);
99  }
100 
101  /// ominus operator
102  static auto ominus(Params const& lhs_params, Params const& rhs_params)
103  -> Tangent {
104  return log((matRx(lhs_params).transpose() * rhs_params).eval());
105  }
106 
107  /// Params examples
108  static auto paramsExamples() -> std::vector<Params> {
109  return std::vector<Params>(
110  {Params::UnitX(), Params::UnitY(), -Params::UnitX(), -Params::UnitY()});
111  }
112 
113  /// Examples of invalid params.
114  static auto invalidParamsExamples() -> std::vector<Params> {
115  return std::vector<Params>(
116  {Params::Zero(), Params::Ones(), 2.0 * Params::UnitX()});
117  }
118 
119  /// Tangent vector examples
120  static auto tangentExamples() -> std::vector<Tangent> {
121  return std::vector<Tangent>({
122  Tangent::Zero(),
123  0.01 * Tangent::UnitX(),
124  0.001 * Tangent::Ones(),
125  });
126  }
127 
128  private:
129  static auto matRx(Params const& params)
130  -> Eigen::Matrix<Scalar, kNumParams, kNumParams> {
131  static Eigen::Vector<TScalar, kDim> const kUnitX =
132  Eigen::Vector<TScalar, kDim>::UnitX();
133  if ((kUnitX - params).squaredNorm() < kEpsilon<Scalar>) {
134  return Eigen::Matrix<Scalar, kNumParams, kNumParams>::Identity();
135  }
136  Params v = params - kUnitX;
137  return Eigen::Matrix<Scalar, kNumParams, kNumParams>::Identity() -
138  2.0 * (v * v.transpose()) / v.squaredNorm();
139  }
140 
141  static auto exp(Tangent const& delta) -> Params {
142  using std::cos;
143  Params params;
144  Scalar theta = delta.norm();
145  params[0] = cos(theta);
146  params.template tail<kDof>() = sinc(theta) * delta;
147  return params;
148  }
149 
150  static auto log(Params const& params) -> Tangent {
151  using std::atan2;
152 
153  static Tangent const kUnitX = Tangent::UnitX();
154  Scalar x = params[0];
155  Tangent tail = params.template tail<kDof>();
156  Scalar theta = tail.norm();
157 
158  if (abs(theta) < kEpsilon<Scalar>) {
159  return atan2(Scalar(0.0), x) * kUnitX;
160  }
161 
162  return (1.0 / theta) * atan2(theta, x) * tail;
163  }
164 
165  static auto sinc(Scalar x) -> Scalar {
166  using std::abs;
167  using std::sin;
168  if (abs(x) < kEpsilon<Scalar>) {
169  return 1.0 - (1.0 / 6.0) * (x * x);
170  }
171  return sin(x) / x;
172  }
173 };

177 /// A n-dimensional vector of unit length.
178 template <class TScalar, int kDim>
179 class UnitVector : public linalg::UnitVectorImpl<TScalar, kDim> {
180  public:
181  /// scalar type
182  using Scalar = TScalar;
183 
184  /// The stateless implementation.
185  using Impl = linalg::UnitVectorImpl<Scalar, kDim>;
186  static_assert(concepts::ManifoldImpl<Impl>);
187 
188  /// The DOF is one dimension smaller than the embedding dimension.
189  /// Examples:
190  /// - A unit circle lives in a 2d space, but a point on a circle can be
191  /// represented using a single scalar (e.g. an angle).
192  /// - A unit sphere lives in a 3d space, but a point on a sphere can be
193  /// described using a 2-tuple (e.g. latitude and longitude).
194  static int constexpr kDof = kDim - 1;
195  /// Internally the unit vector is represented by a vector, hence the number of
196  /// parameters equals the number of dimension of the embedding space.
197  static int constexpr kNumParams = kDim;
198 
199  /// internal parameter representation
200  using Params = Eigen::Vector<Scalar, kNumParams>;
201  /// tangent vector representation
202  using Tangent = Eigen::Vector<Scalar, kDof>;
203 
204  /// Tries to construct it given an input vector v.
205  ///
206  /// Returns error if the norm of v is not close to 1
207  static auto tryFromUnitVector(Eigen::Matrix<TScalar, kDim, 1> const& v)
208  -> Expected<UnitVector> {
209  SOPHUS_TRY(auto, maybe_valid, Impl::areParamsValid(v));
210  UnitVector unit_vector;
211  unit_vector.vector_ = v;
212  return unit_vector;
213  }
214 
215  /// Constructs it given an input vector v.
216  ///
217  /// Precondition: norm of v must be close to 1
218  static auto fromParams(Eigen::Matrix<TScalar, kDim, 1> const& v)
219  -> UnitVector {
220  Expected<UnitVector> e_vec = tryFromUnitVector(v);
221  if (!e_vec.has_value()) {
222  SOPHUS_PANIC("{}", e_vec.error());
223  }
224  return e_vec.value();
225  }
226 
227  /// Precondition: v must be of unit length.
228  static auto fromUnitVector(Eigen::Matrix<TScalar, kDim, 1> const& v)
229  -> UnitVector {
230  return fromParams(v);
231  }
232 
233  /// Constructs it given an input vector v.
234  ///
235  /// If v is not of unit length then it will be normalized accordingly.
236  static auto fromVectorAndNormalize(Eigen::Matrix<TScalar, kDim, 1> const& v)
237  -> UnitVector {
238  return fromUnitVector(v.normalized());
239  }
240 
241  /// oplus operator
242  auto oplus(Tangent const& delta) const -> UnitVector {
243  UnitVector v;
244  v.vector_ = Impl::oplus(vector_, delta);
245  return v;
246  }
247 
248  /// ominus operator
249  auto ominus(UnitVector const& rhs_params) const -> Tangent {
250  return Impl::ominus(vector_, rhs_params.vector_);
251  }
252 
253  /// Set state using params.
254  void setParams(Eigen::Matrix<TScalar, kDim, 1> const& v) const {
255  SOPHUS_ASSERT(Impl::areParamsValid(v));
256  vector_.params = v;
257  }
258 
259  /// Get params
260  [[nodiscard]] auto params() const -> Eigen::Matrix<TScalar, kDim, 1> const& {
261  return vector_;
262  }
263 
264  /// Return as ordinary Eigen vector
265  [[nodiscard]] auto vector() const -> Eigen::Matrix<TScalar, kDim, 1> const& {
266  return vector_;
267  }
268 
269  /// unsafe mutable access to params pointer
270  ///
271  /// Note: It is the users responsibility to hold up the class invariant that
272  /// the vector must have unit length.
273  [[nodiscard]] auto unsafeMutPtr() { return this->vector_.data(); }
274 
275  /// cost access to params pointer
276  [[nodiscard]] auto ptr() const { return this->vector_.data(); }
277 
278  /// Copy constructor.
279  UnitVector(UnitVector const&) = default;
280  /// Copy assignment.
281  auto operator=(UnitVector const&) -> UnitVector& = default;
282 
283  /// Average a set of unit vectors.
284  ///
285  /// This is done by summing uo the vectors and re-normalizing the result.
286  template <concepts::Range TSequenceContainer>
287  static auto average(TSequenceContainer const& range) -> UnitVector {
288  size_t const len = std::distance(std::begin(range), std::end(range));
289  SOPHUS_ASSERT_GE(len, 0);
290 
291  Params params = Params::Zero();
292  for (auto const& m : range) {
293  params += m.params();
294  }
295  return fromVectorAndNormalize(params / len);
296  }
297 
298  private:
299  UnitVector() {}
300 
301  // Class invariant: v_ is of unit length.
302  Eigen::Matrix<TScalar, kDim, 1> vector_;
303 };
